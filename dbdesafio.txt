create database dbdesafio
default character set utf8
default collate utf8_general_ci;

use dbdesafio;

create table bancos (
id int not null auto_increment,
instituicao varchar(20),
tipo varchar(20),
saldo decimal(6,2),
primary key (id)
) default charset = utf8;

create table receitas (
id int not null auto_increment,
valor decimal(6,2),
dataRecebimento date,
dataRecebimentoEsperado date,
descricao varchar(20),
conta varchar(10),
tipoReceita varchar(10),
primary key (id)
) default charset = utf8;

create table despesas (
id int not null auto_increment,
valor decimal(6,2),
dataPagamento date,
dataPagamentoEsperado date,
tipoDespesa varchar(10),
conta varchar(10),
primary key (id)
) default charset = utf8;
